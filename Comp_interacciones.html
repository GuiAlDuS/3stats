Interacciones entre variables y su efecto en predicciones

La formula del proceso generador de datos:

\[ Y = \alpha + \beta_1A + \beta_2B + \beta_3AB + \epsilon \]

Creamos nuestros datos:

set.seed(777)

n = 100

alfa <- 7
beta_1 <- 0.3
beta_2 <- -1.5
beta_3 <- 3

A <- runif(n, min = 11, max = 45)
B <- runif(n, min = 0, max =  15)
e = rnorm(n, mean = 0, sd=0.2)

Y <- alfa + beta_1*A + beta_2*B + beta_3*A*B + e

df <- data.frame(cbind(A, B, Y))

pred_lgm_AyB <- predict(mod_lgb, matrix(c(A1, B1), ncol = 2))

## [1] 725.4669

Notese que a pesar de no haber explicitamente creado la interaccin entre las variables, el ensamble de rboles logr darnos una prediccin cercana al valor real.

Ahora veamos el valor que nos predice nuestra regresin que s tom en cuenta la interaccin:

pred_reg2 = predict(mod2, 
                    data.frame(A = A1, B = B1))
pred_reg2
##       1 
## 693.414

Nos da un valor muy cercano al verdadero.

Y ahora, qu tal si ponemos la interaccin en las variables que vamos a utilizar con el RF?

AB <- A*B

df_2 <- cbind(df, AB)

m2_rf <- ranger(
  formula =  Y ~ A + B + AB,
  data    = df_2,
  importance = "permutation", 
  local.importance = TRUE
)

pred_rf_AB <- predict(m2_rf, 
                       data.frame(A = A1, B = B1, AB = A1*B1))$predictions
pred_rf_AB

Calculemos ahora las predicciones para una serie de valores:

n2 <- 1000

An <- runif(n2, min = 15, max = 32)
Bn <- runif(n2, min = 0, max =  8)


Yn_true <- alfa + beta_1*An + beta_2*Bn + beta_3*An*Bn# + e[1]

df_n <- data.frame(cbind(Yn_true, An, Bn))

rf2_1n <- predict(m2_rf, data.frame(A=An, B=Bn, AB=An*Bn))

lgb2_1n <- predict(mod_lgb, matrix(c(An, Bn), ncol=2))

list_mods <- list(mod1, mod_rf_1, mod2, m1_dt)

test <- lapply(list_mods, function(x) predict(x, data.frame(A=df_n$An, B=df_n$Bn)))

df_results <- data.frame(
  Yn_true = Yn_true,
  reg.lin.AyB = test[[1]],
  rf_AyB = test[[2]][["predictions"]],
  lgb_AyB = lgb2_1n,
  reg.lin.AyByAB = test[[3]],
  rf_AyByAB = rf2_1n$predictions,
  arbol = test[[4]]
)

library(data.table)
setDT(df_results)

df_results_difs <- df_results[, 
  .(`reg.lin.AyB` = abs(`reg.lin.AyB` - Yn_true),
    rf_AyB = abs(rf_AyB - Yn_true),
    lgb_AyB = abs(lgb_AyB - Yn_true),
    `reg.lin.AyByAB` = abs(`reg.lin.AyByAB` - Yn_true),
    rf_AyByAB = abs(rf_AyByAB - Yn_true),
    arbol = abs(arbol - Yn_true))
]

summary <- data.frame(sapply(df_results_difs, function(x) round(mean(x), 2)))

names(summary)[1]<- 'promedio'
