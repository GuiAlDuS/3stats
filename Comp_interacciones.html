La formula del proceso generador de datos:

\[ Y = \alpha + \beta_1A + \beta_2B + \beta_3AB + \epsilon \]

Creamos nuestros datos:

set.seed(777)

n = 100

alfa <- 7
beta_1 <- 0.3
beta_2 <- -1.5
beta_3 <- 3

A <- runif(n, min = 11, max = 45)
B <- runif(n, min = 0, max =  15)
e = rnorm(n, mean = 0, sd=0.2)

Y <- alfa + beta_1*A + beta_2*B + beta_3*A*B + e

df <- data.frame(cbind(A, B, Y))

n2 <- 1000

An <- runif(n2, min = 15, max = 32)
Bn <- runif(n2, min = 0, max =  8)


Yn_true <- alfa + beta_1*An + beta_2*Bn + beta_3*An*Bn# + e[1]

df_n <- data.frame(cbind(Yn_true, An, Bn))

rf2_1n <- predict(m2_rf, data.frame(A=An, B=Bn, AB=An*Bn))

lgb2_1n <- predict(mod_lgb, matrix(c(An, Bn), ncol=2))

list_mods <- list(mod1, mod_rf_1, mod2, m1_dt)

test <- lapply(list_mods, function(x) predict(x, data.frame(A=df_n$An, B=df_n$Bn)))

df_results <- data.frame(
  Yn_true = Yn_true,
  reg.lin.AyB = test[[1]],
  rf_AyB = test[[2]][["predictions"]],
  lgb_AyB = lgb2_1n,
  reg.lin.AyByAB = test[[3]],
  rf_AyByAB = rf2_1n$predictions,
  arbol = test[[4]]
)

library(data.table)
setDT(df_results)

df_results_difs <- df_results[, 
  .(`reg.lin.AyB` = abs(`reg.lin.AyB` - Yn_true),
    rf_AyB = abs(rf_AyB - Yn_true),
    lgb_AyB = abs(lgb_AyB - Yn_true),
    `reg.lin.AyByAB` = abs(`reg.lin.AyByAB` - Yn_true),
    rf_AyByAB = abs(rf_AyByAB - Yn_true),
    arbol = abs(arbol - Yn_true))
]

summary <- data.frame(sapply(df_results_difs, function(x) round(mean(x), 2)))
